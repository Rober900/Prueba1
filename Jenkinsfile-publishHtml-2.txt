pipeline {
  agent any
  options {
    buildDiscarder(logRotator(numToKeepStr: '5'))
  }
  stages {
    stage('Build') {
      steps {
        sh './mvnw clean install site surefire-report:report'
        sh './mvnw clean package'
        java -jar ./TestSmellDetector-0.1-jar-with-dependencies.jar Ejemplo.csv
      }
    }
  }
  post {
    success {
      publishHTML([allowMissing: false, alwaysLinkToLastBuild: false, keepAll: false, reportDir: 'target/site', reportFiles: 'Output.html', reportName: 'Output Report', reportTitles: '', useWrapperFileDirectly: true])
    }
  }
}
